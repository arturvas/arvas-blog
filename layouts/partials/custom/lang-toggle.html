{{/* Language toggle (PT | EN). Behavior: - On the homepage and archives page, always rendered. - On a single post, only rendered if a translation exists (.IsTranslated). - Clicking a side sets the lang_pref cookie and navigates to the equivalent URL in that language. For posts without a translation, the EN side is hidden entirely (we never link to a non-existent translation). */}} {{- $isHome := .IsHome -}} {{- $relPerm := .RelPermalink -}} {{- $isArchives := or (eq $relPerm "/archives/") (eq $relPerm "/en/archives/") -}} {{- $isOffTopic := or (eq $relPerm "/off-topic/") (eq $relPerm "/en/off-topic/") -}} {{- $isPost := .IsPage -}} {{- $hasTranslations := gt (len .Translations) 0 -}} {{- $showOnPost := and $isPost $hasTranslations -}} {{- $show := or $isHome $isArchives $isOffTopic $showOnPost -}} {{- if $show -}} {{- $currentLang := .Site.Language.Lang -}} {{/* Resolve the URL for each language */}} {{- $ptUrl := "/" -}} {{- $enUrl := "/en/" -}} {{- if $isPost -}} {{- if eq $currentLang "pt-br" -}} {{- $ptUrl = .RelPermalink -}} {{- range .Translations -}} {{- if eq .Lang "en" -}}{{- $enUrl = .RelPermalink -}}{{- end -}} {{- end -}} {{- else -}} {{- $enUrl = .RelPermalink -}} {{- range .Translations -}} {{- if eq .Lang "pt-br" -}}{{- $ptUrl = .RelPermalink -}}{{- end -}} {{- end -}} {{- end -}} {{- else if $isArchives -}} {{- $ptUrl = "/archives/" -}} {{- $enUrl = "/en/archives/" -}} {{- else if $isOffTopic -}} {{- $ptUrl = "/off-topic/" -}} {{- $enUrl = "/en/off-topic/" -}} {{- end -}}
PT | EN
{{- end -}}
